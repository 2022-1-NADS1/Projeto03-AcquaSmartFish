#include <ESP8266WiFi.h>
#include <WiFiClient.h>
#include <ESP8266WebServer.h>
#include <ESP8266mDNS.h>
#include "index_html.h"
#define D1 5

ESP8266WebServer server(80);

String potinho = "";
int estado = D1;

void setup() {
Serial.begin(115200);
pinMode (5, INPUT);
pinMode (4, OUTPUT);

WiFi.mode(WIFI_STA);
WiFi.begin("Conversas Fiadas","naocuramressaca");
Serial.println("");
// espera a conexao
while (WiFi.status() != WL_CONNECTED) {
delay(500);
Serial.print(".");
}
Serial.println("");
Serial.print("Conectado ");
Serial.println(WiFi.localIP());

server.on("/",Chamaeu);
server.on("/LED",ChamaLed);
server.begin();
pinMode(LED_BUILTIN, OUTPUT);
//pinMode(A0, OUTPUT);
}
void Chamaeu(){
//server.send(200,"text/plain", "ola");

String page=INDEX_HTML + potinho + INDEX_HTMLfinal;

server.send(200,"text/html", page);
}
void ChamaLed(){
digitalWrite(LED_BUILTIN, HIGH);
}
void loop() {
server.handleClient();

if (digitalRead(5)==HIGH)
{
  potinho = "Potinho vazio";
  Serial.println("Estou recebendo sinal da D1");
  digitalWrite(4, HIGH);
}
if (digitalRead(5)==LOW)
{
  potinho = "Potinho cheio";
  Serial.println("N√ÉO estou recebendo sinal da D1");
  digitalWrite(4, LOW);
}
delay(1000);
}
